1) You can pass a record to `initialValues` and it will pass through the underlying data to the form

2) final-form expects a function as a child which you then render a `<form>` element with and pass through
   handleSubmit. Most the time this is the same thing so you optionally just pass through renderable
   children and the form will be created implicitly for you.


import { Form } from '@prestojs/final-form';
import { User } from 'docSite/models';
import { PropTypes } from 'docSite/components/View';

<View
    componentName="Form"
    componentProps={{
        layout: {
            type: PropTypes.String,
            value: 'horizontal',
        },
        onSubmit: {
            type: PropTypes.Function,
            value: 'data=> console.log(data)',
        },
        children: {
            value: `
<Form.Item field={User.fields.firstName} />
<Form.Item field={User.fields.lastName} />
<Form.Item field={User.fields.email} />
`,
            type: PropTypes.ReactNode,
            description: 'Visible label.',
        },
    }}
    scope={{
        Form,
        User,
    }}
    imports={{
        '@prestojs/final-form': {
            named: ['Form'],
        },
    }}
/>
